perl -n -e 'if (/^\d\d:\d\d\.\d+/) {$event =~ s/.\n/<line>/g; print $event."\n"; $event = "";} $event .= $_; END{print $event};' | \
perl -pe 's/\xef\xbb\xbf//g' | \
grep -P ".*TTIMEOUT" | \
perl -pe 's/^\d+:\d+.\d+-//g' | \
perl -pe 's/.*TTIMEOUT,.*Context=/,Context=/g' | \
perl -pe 's/<line>//g' | \
awk -F',Context=' '{count[$2]+=1;} END {for(i in count) {print count[i] " " i}}' | \
sort -rnb | \
head -n 10
